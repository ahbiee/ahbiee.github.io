#include <bits/stdc++.h>
using namespace std;
using ll = long long;

vector<ll> arr, sum, add;

void up(int i){
    sum[i] = sum[i*2] + sum[i*2+1];
}

void lazy(int i, ll v, int n){
    sum[i] += v*n;
    add[i] += v;
}

void down(int i, int ln, int rn){
    if(add[i] != 0){
        lazy(i*2, add[i], ln);
        lazy(i*2+1, add[i], rn);
        add[i] = 0;
    }
}

void build(int l, int r, int i){
    if(l == r) sum[i] = arr[l];
    else{
        int mid = (l+r)/2;
        build(l, mid, i*2);
        build(mid+1, r, i*2+1);
        up(i);
    }
    add[i] = 0;
}

void update(int jobl, int jobr, ll jobv, int l, int r, int i){
    if(jobl <= l && r <= jobr) lazy(i, jobv, r-l+1);
    else{
        int mid = (l+r)/2;
        down(i, mid-l+1, r-mid);
        if(jobl <= mid) update(jobl, jobr, jobv, l, mid, i*2);
        if(jobr > mid) update(jobl, jobr, jobv, mid+1, r, i*2+1);
        up(i);
    }
}

ll query(int jobl, int jobr, int l, int r, int i){
    if(jobl <= l && r <= jobr) return sum[i];
    int mid = (l+r)/2;
    down(i, mid-l+1, r-mid);
    ll total = 0;
    if(jobl <= mid) total += query(jobl, jobr, l, mid, i*2);
    if(jobr > mid) total += query(jobl, jobr, mid+1, r, i*2+1);
    return total;
}

int main() {
    int n, m;
    cin >> n >> m;
    arr.assign(n+1, 0);
    sum.assign(4*(n+1), 0);
    add.assign(4*(n+1), 0);

    for(int i=1; i<=n; ++i) cin >> arr[i];

    build(1, n, 1);

    int op;
    ll x, y, k;
    while(m--){
        cin >> op;
        if(op == 1){
            cin >> x >> y >> k;
            update(x, y, k, 1, n, 1);
        }
        else{
            cin >> x >> y;
            cout << query(x, y, 1, n, 1) << '\n';
        }
    }
    return 0;
}
