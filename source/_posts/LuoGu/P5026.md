#include <bits/stdc++.h>
#define pb push_back
#define eb emplace_back
#define mp make_pair
#define INF 0x3f3f3f3f
#define all(x) (x).begin(), (x).end()

using namespace std;
using ll = long long;
using pii = pair<int, int>;

#define MAXN 1000005
#define OFFSET 30005

int arr[OFFSET + MAXN + OFFSET];
int n, m;
int v, x;

void build(int l, int r, int s, int e, int d){
    arr[l+OFFSET] += s;
    arr[l+1+OFFSET] += d-s;
    arr[r+1+OFFSET] -= d+e;
    arr[r+2+OFFSET] += e;
}

void fall(){
    build(x-3*v+1, x-2*v, 1, v, 1);
    build(x-2*v+1, x, v-1, -v, -1);
    build(x+1, x+2*v, -v+1, v, 1);
    build(x+2*v+1, x+3*v-1, v-1, 1, -1);
}

void calculate(){
    for(int i=1; i<=m+OFFSET; ++i){
        arr[i] += arr[i-1];
    }
    for(int i=1; i<=m+OFFSET; ++i){
        arr[i] += arr[i-1];
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    // n個朋友，湖長度為m
    while(cin >> n >> m){
        while(n--){
            cin >> v >> x;
            fall();
        }
        calculate();
        cout << arr[1+OFFSET];
        for(int i=2; i<=m; ++i){
            cout << " " << arr[i+OFFSET];
        }
        cout << '\n';
    }
    return 0;
}
