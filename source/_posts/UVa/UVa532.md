#include <bits/stdc++.h>
using namespace std;

int l, r, c;
int dx[6] = {1, -1, 0, 0, 0, 0};
int dy[6] = {0, 0, 1, -1, 0, 0};
int dz[6] = {0, 0, 0, 0, 1, -1};

struct pos{
    int l, r, c;
    bool operator==(const pos &o) const{
        if(l != o.l) return false;
        if(r != o.r) return false;
        if(c != o.c) return false;
        return true;
    }
};

char v[35][35][35];
bool visited[35][35][35];
pos st, ed;

void bfs(){
    queue<pair<pos, int>> q; // 紀錄位置與抵達時的時間
    q.push({st, 0});
    bool found = false;
    while(!q.empty()){
        pair<pos, int> cur = q.front(); q.pop();
        if(visited[cur.first.l][cur.first.r][cur.first.c]) continue;
        visited[cur.first.l][cur.first.r][cur.first.c] = true;
        if(cur.first == ed){
            printf("Escaped in %d minute(s).\n", cur.second);
            found = true;
            return;\
        }

        for(int i=0; i<6; ++i){
            int nx = cur.first.l + dx[i];
            int ny = cur.first.r + dy[i];
            int nz = cur.first.c + dz[i];
            if(nx <= 0 || nx > l) continue;
            if(ny <= 0 || ny > r) continue;
            if(nz <= 0 || nz > c) continue;
            if(visited[nx][ny][nz] || v[nx][ny][nz] == '#') continue;
            q.push({{nx, ny, nz}, cur.second+1});
        }
    }
    if(!found) printf("Trapped!\n");
}

int main() {
    while(~scanf("%d%d%d", &l, &r, &c)){
        if(l == 0 && r == 0 && c == 0) break;
        for(int i=1; i<=l; ++i){
            for(int j=1; j<=r; ++j){
                for(int k=1; k<=c; ++k){
                    scanf(" %c", &v[i][j][k]);
                    if(v[i][j][k] == 'S') st = {i, j, k};
                    if(v[i][j][k] == 'E') ed = {i, j, k};
                    visited[i][j][k] = false;
                }
            }
        }
        bfs();
    }
    return 0;
}
