#include <bits/stdc++.h>
using namespace std;
using ll = long long;

ll count_cnt(ll bill){
    if(bill > 4979900) return 1000000 + (bill-4979900)/7;
    if(bill > 29900) return 10000 + (bill-29900)/5;
    if(bill > 200) return 100 + (bill-200)/3;
    return bill/2;
}

ll count_bill(ll cnt){
    if(cnt <= 100) return cnt*2;
    if(cnt <= 10000) return 200 + (cnt-100)*3;
    if(cnt <= 1000000) return 29900 + (cnt-10000)*5;
    return 4979900 + (cnt-1000000)*7;
}

ll solve(ll cnt, ll goal){
    ll low = 0, high = cnt/2;
    while(low <= high){
        ll mid = (high + low) / 2;
        ll diff = count_bill(cnt-mid) - count_bill(mid);
        if(diff == goal) return mid;
        else if(diff < goal) high = mid-1;
        else low = mid + 1;
    }
    return -1;
}

int main() {
    ll together, diff;
    while(scanf("%lld%lld", &together, &diff) && together){
        ll cnt = count_cnt(together);
        printf("%lld\n", count_bill(solve(cnt, diff)));
    }
    return 0;
}
