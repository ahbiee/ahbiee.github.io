#include <bits/stdc++.h>
using namespace std;

struct point{
    double l, r; // left, right after calculation
    bool operator<(const point &o) const{
        if(l != o.l) return l < o.l; // sort by begin time
        return r < o.r;
    }
};

int main(){
    int n, l, w;
    while(cin >> n >> l >> w){
        vector<point> v;
        double pos, rad;
        for(int i=0; i<n; ++i){
            cin >> pos >> rad;
            double real;
            if(rad < w/2.0) continue;
            real = sqrt(rad*rad - (w/2.0)*(w/2.0));
            v.push_back({pos-real, pos+real});
        }
        sort(v.begin(), v.end());

        int cnt = 0;
        int i = 0;
        double furthest = 0.0;
        bool noAns = false;

        while(furthest < l){
            double cur = furthest;
            while(i < v.size() && v[i].l <= furthest){
                cur = max(cur, v[i].r);
                cout << "cur: " << cur << '\n';
                ++i;
            }
            if(cur - furthest < 1e-8){
                noAns = true;
                break;
            }
            furthest = cur;
            ++cnt;
        }
        if(noAns) cout << -1 << '\n';
        else cout << cnt << '\n';
    }
    return 0;
}
