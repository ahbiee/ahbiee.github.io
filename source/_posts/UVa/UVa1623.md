#include <bits/stdc++.h>
#define MAXN 1000005
using namespace std;

vector<int> a(MAXN), ans(MAXN), f(MAXN), lake(MAXN);

int n, m;

int find(int x){
    if(f[x] == x) return x;
    return f[x] = find(f[x]);
}

bool solve(){
    int x;
    for(int i=1; i<=n; ++i) lake[i] = 0;
    for(int i=1; i<=m; ++i){
        if(a[i] == 0) continue; // 要找有下雨的日子
        x = find(lake[a[i]]);
        if(x <= i){
            ans[x] = a[i];
            f[x] = find(x+1);
        }
        else return false;
        lake[a[i]] = i;
    }
    printf("YES\n");
    for(int i=1; i<=m; ++i) if(a[i] == 0) printf("%d ", ans[i]);
    printf("\n");
    return true;
}

int main(){
    int kase;
    cin >> kase;
    while(kase--){
        cin >> n >> m;
        for(int i=1; i<=m; ++i){
            cin >> a[i];
            ans[i] = 0;
        }
        int last = m+1;
        f[m+1] = m+1;
        for(int i=m; i>=0; --i){
            if(a[i] == 0 && i != 0) last = i;
            f[i] = last;
        }

        if(!solve()) printf("NO\n");
    }
    return 0;
}
