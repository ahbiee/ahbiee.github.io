#include <bits/stdc++.h>

using namespace std;

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);

    int kase;

    string password;
    string guess;
    while(cin >> kase){
        if(kase == -1) break;
        cin >> ws;
        getline(cin, password);
        getline(cin, guess);

        unordered_set<char> s, guessed;
        for(char c : password) s.insert(c);

        int wrong = 0;
        int all = s.size(), correct = 0;

        for(char current : guess){
            if(wrong == 7 || correct == all) break;
            if(guessed.count(current)) continue;
            guessed.insert(current);
            if(s.count(current)){
                correct++;
                s.erase(current);
            }
            else{
                wrong++;
            }
            
        }

        cout << "Round " << kase << '\n';
        if(wrong == 7){ // lose
            cout << "You lose.\n";
        }
        else if(s.empty()){ // win
            cout << "You win.\n";
        }
        else{ // chickened out
            cout << "You chickened out.\n";
        }
    }

    return 0;
}
