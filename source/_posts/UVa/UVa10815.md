using namespace std;
using ll = long long;
using pii = pair<int, int>;

set<string> s;

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    char c;
    string word = "";
    while(cin.get(c)){
        if(isalpha(c)) word += tolower(c);
        else{
            if(word != "") s.insert(word);
            word = "";
        }
    }
    for(auto& str : s) cout << str << '\n';
    return 0;
}
