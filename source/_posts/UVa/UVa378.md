#include <bits/stdc++.h>
using namespace std;

const double eps = 1e-9;

struct Point{
    double x, y;
};

struct Line{
    Point pos, vector;
};

double cross(Point a, Point b){
    return a.x*b.y - a.y*b.x;
}

void solve(Line &u, Line &v){
    Point vec = {v.pos.x - u.pos.x, v.pos.y - u.pos.y}; // 兩點間的向量
    double crossProduct = cross(u.vector, v.vector);

    if(fabs(crossProduct) < eps){
        if(fabs(cross(vec, u.vector)) < eps) printf("LINE\n");
        else printf("NONE\n");
    }
    else{
        double t = cross(vec, v.vector) / crossProduct;
        double intersect_x = u.pos.x + t*u.vector.x;
        double intersect_y = u.pos.y + t*u.vector.y;
        printf("POINT %.2f %.2f\n", intersect_x, intersect_y);
    }
}

int main() {
    int n;
    scanf("%d", &n);
    Point a, b, c, d;
    printf("INTERSECTING LINES OUTPUT\n");
    while(n--){
        scanf("%lf%lf%lf%lf%lf%lf%lf%lf", &a.x, &a.y, &b.x, &b.y, &c.x, &c.y, &d.x, &d.y);
        Line u, v;
        u.pos = a;
        u.vector = {b.x-a.x, b.y-a.y};
        v.pos = c;
        v.vector = {d.x-c.x, d.y-c.y};
        solve(u, v);
    }
    printf("END OF OUTPUT\n");
    return 0;
}
