```cpp=
#include <bits/stdc++.h>
using namespace std;

struct pos{
    int l, r;
    bool operator<(const pos&o) const{
        return l < o.l;
    }
};

int main() {
    int len, gas;
    int x, r;

    while(cin >> len >> gas && len && gas){
        vector<pos> v;
        for(int i=0; i<gas; ++i){
            cin >> x >> r;
            v.push_back({x-r, x+r});
        }
        sort(v.begin(), v.end());

        int i=0;
        int farthest = 0;
        int cnt = 0;
        bool noAns = false;
        while(farthest < len){
            int cur = farthest;
            while(i < gas && v[i].l <= farthest){
                cur = max(cur, v[i].r);
                ++i;
            }
            if(cur == farthest){
                noAns = true;
                break;
            }
            farthest = cur;
            ++cnt;
        }
        if(noAns) cout << -1 << '\n';
        else cout << gas-cnt << '\n';
    }
    return 0;
}
```
