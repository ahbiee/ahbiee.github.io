#include <bits/stdc++.h>
#define INF 0x3f3f3f3f
#define MAXN 105
using namespace std;

int n, e, t, m; 
// cells, exits, time, connections

struct Edge{
    int goal, weight;
    bool operator<(const Edge &o) const{
        return weight > o.weight;
    };
};

vector<Edge> g[MAXN];
vector<int> dist; // distance from e to i

void init(){
    for(int i=0; i<MAXN; ++i){
        g[i].clear();
    }
    dist.assign(MAXN, INF);
}

int solve(){
    priority_queue<Edge> pq;
    vector<bool> visited(MAXN, false);

    dist[e] = 0;
    pq.push({e, 0});

    Edge cur;
    while(!pq.empty()){
        cur = pq.top(); pq.pop();
        int start = cur.goal;
        if(visited[start]) continue;
        visited[start] = true;

        for(Edge &ed : g[start]){
            int goal = ed.goal;
            int weight = ed.weight;
            if(dist[goal] > dist[start] + weight){
                dist[goal] = dist[start] + weight;
                pq.push({goal, dist[goal]});
            }
        }
    }
    int cnt = 0;
    for(int i=1; i<=n; ++i) if(dist[i] <= t) ++cnt;
    return cnt;
}

int main() {
    ios::sync_with_stdio(0); cin.tie(0);
    int kase;
    cin >> kase;

    int a, b, d;
    while(kase--){
        init();
        cin >> n >> e >> t >> m;
        for(int i=0; i<m; ++i){
            cin >> a >> b >> d;
            g[b].push_back({a, d});
        }
        int ans = solve();
        cout << ans << '\n';
        if(kase) cout << '\n';
    }
    return 0;
}
