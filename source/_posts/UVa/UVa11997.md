#include <bits/stdc++.h>
#define pb push_back
#define eb emplace_back
#define mp make_pair
#define INF 0x3f3f3f3f
#define all(x) (x).begin(), (x).end()

using namespace std;
using ll = long long;

#define MAXN 755

int k;
int a[MAXN], b[MAXN];

struct Item{
    int sum, idx; // 當前總和, 是跟B的第幾個idx相加
    Item(int sum, int idx) : sum{sum}, idx{idx} {}
    bool operator < (const Item &o) const{
        return sum > o.sum;
    }
};

void merge(int a[], int b[], int dest[]){
    priority_queue<Item> pq;
    for(int i=0; i<k; ++i) pq.push({a[i] + b[0], 0});
    for(int i=0; i<k; ++i){
        Item curr = pq.top();
        pq.pop();

        dest[i] = curr.sum;
        int idx = curr.idx;
        if(idx + 1 < k) pq.push({curr.sum - b[idx] + b[idx+1], idx+1});
    }
}

int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    while(cin >> k){
        for(int i=0; i<k; ++i) cin >> a[i];
        sort(a, a+k);

        for(int row=2; row<=k; ++row){
            for(int i=0; i<k; ++i) cin >> b[i];
            sort(b, b+k);

            merge(a, b, a);
        }

        for(int i=0; i<k; ++i) cout << (i ? " " : "") << a[i];
        cout << endl;
    }
    return 0;
}
