```cpp=
class Solution {
public:
    vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
        vector<int> arr(n+2);
        for(vector<int>& books : bookings){
            arr[books[0]] += books[2];
            arr[books[1]+1] -= books[2];
        }

        vector<int> ans;
        for(int i=1; i<n+1; ++i){ // index from 1 ~ n
            arr[i] += arr[i-1];
            ans.push_back(arr[i]);
        }
        return ans;
    }
};
```
