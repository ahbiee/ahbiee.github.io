```cpp=
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int findPeakElement(vector<int>& nums) {
        if (nums.size() == 1) return 0;
        int n = nums.size();
        if(nums[0] > nums[1]) return 0;
        if(nums[n-1] > nums[n-2]) return n-1;
        int l = 1, r = n-2, m = -1, ans = -1;
        while(l <= r){
            m = l + ((r-l) >> 1);
            if(nums[m] < nums[m-1]){
                r = m-1;
            }
            else if(nums[m] < nums[m+1]){
                l = m+1;
            }
            else{
                ans = m;
                break;
            }
        }
        return ans;
    }
};
```
