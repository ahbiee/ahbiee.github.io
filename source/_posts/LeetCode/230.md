```cpp=
class Solution {
public:
    // l < m < r
    int ans;
    int i;
    int kthSmallest(TreeNode* root, int k) {
        ans = 0 ;
        i = 0 ;
        inorder(root, k);
        return ans;
    }

    void inorder(TreeNode* cur, int k){
        if(cur->left != nullptr) inorder(cur->left, k);
        if(++i == k){
            ans = cur->val;
            return;
        }
        if(cur->right != nullptr) inorder(cur->right, k);
    }
};
```
