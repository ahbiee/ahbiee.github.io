題目:
https://leetcode.com/problems/minimum-size-subarray-sum/

<!-- more -->

```cpp=
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int total = 0;
        int len = 0x3f3f3f3f;
        for(int l=0, r=0; r<nums.size(); ++r){
            total += nums[r];
            while(total >= target){
                len = min(len, r-l+1);
                total -= nums[l++];
            }
        }
        return len == 0x3f3f3f3f ? 0 : len;
    }
};
```
