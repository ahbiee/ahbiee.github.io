```cpp=
class Solution {
public:
    int dp[50];
    int climbStairs(int n) {
        memset(dp, -1, sizeof(dp));
        return solve(0, n);
    }

    int solve(int cur, int n){
        if(cur == n) return 1;
        if(cur > n) return 0;
        if(dp[cur] != -1) return dp[cur];
        dp[cur] = solve(cur+1, n) + solve(cur+2, n);
        return dp[cur];
    }
};
```
