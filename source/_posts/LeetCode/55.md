```cpp=
class Solution {
public:
    bool canJump(vector<int>& nums) {
        if(nums.size() <= 1) return true;
        int idx = 0;
        int max_jump = nums[idx];
        while(max_jump > 0 && idx < nums.size()-1){
            max_jump = max(--max_jump, nums[++idx]);
        }
        if(idx == nums.size()-1) return true;
        else return false;
    }
};
```
