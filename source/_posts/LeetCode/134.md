```cpp=
class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int Size = gas.size();

        for(int l=0, r=0, sum; l < Size; l = r+1, r=l){
            sum = 0;
            while(sum + gas[r%Size] - cost[r%Size] >= 0){
                if(r-l+1 == Size) return l;
                sum += gas[r%Size] - cost[r%Size];
                ++r;
            }
        }
        return -1;
    }
};
```
