```cpp=
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int buy = prices[0];
        int diff = 0;
        for(int i=1; i<prices.size(); ++i){
            buy = min(buy, prices[i]);
            diff = max(diff, prices[i]-buy);
        }
        return diff;
    }
};
```
